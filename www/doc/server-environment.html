const char* parsegraph_nameEnvironmentStatus(parsegraph_EnvironmentStatus)

Returns a string that describes the given environment status.

parsegraph_prepareEnvironmentStatements(pool, dbd)

Loads the Environment module's strings into the dbd prepared statements hash.
  • Returns parsegraph_Environment_INTERNAL_ERROR if a statement failed to prepare.
  • Returns parsegraph_Environment_OK if all statements prepared successfully.

parsegraph_upgradeEnvironmentTables(pool, dbd)

Creates and upgrades the database's environment tables.
  • Returns parsegraph_Environment_INTERNAL_ERROR if the environment table could not be created or updated.
  • Returns parsegraph_Environment_OK if an updated environment table now exists.

parsegraph_createEnvironment(pool, dbd, parsegraph_GUID* createdEnv)

Creates a new environment, returning its GUID in the given GUID pointer.
  • Returns parsegraph_Environment_OK if the environment was created.
  • Returns parsegraph_Environment_INTERNAL_ERROR if an internal error prevented creation.

parsegraph_cloneEnvironment(pool, dbd, parsegraph_GUID* clonedEnv, parsegraph_GUID* createdEnv)

Clones the specified environment, returning the GUID of the new environment in the specified GUID pointer.
  • Returns parsegraph_Environment_OK if the environment was cloned.
  • Returns parsegraph_Environment_INTERNAL_ERROR if an internal error prevented the environment from being cloned.
  • Returns parsegraph_Environment_CLONE_UNSUPPORTED if the environment's content does not support cloning.
  • Returns parsegraph_Environment_NOT_FOUND if the given GUID does not specify an environment.

parsegraph_destroyEnvironment(pool, dbd, parsegraph_GUID* targetEnv)

Destroys the environment specified by the given GUID.
  • Returns parsegraph_Environment_OK if the environment was destroyed.
  • Returns parsegraph_Environment_INTERNAL_ERROR if an internal error prevented the environment from being deleted.

parsegraph_getEnvironmentGUIDForId(pool, dbd, int environmentId, parsegraph_GUID* envGuid)

Retrieves using envGuid the GUID of the environment specified by environmentId.

parsegraph_getEnvironmentIdForGUID(pool, dbd, parsegraph_GUID* envGuid, int* environmentId)

Retrieves using environmentId the ID of the environment specified by envGuid.

parsegraph_getOnlineEnvironment(pool, dbd, struct parsegraph_user_login* userLogin, parsegraph_GUID* onlineEnv)

Retrieves using onlineEnv the online environment for the specified user login.
  • Returns parsegraph_Environment_OK on success.
  • Returns parsegraph_Environment_NOT_FOUND if the session has no online environment.

parsegraph_saveEnvironment(pool, dbd, int userId, apr_dbd_results_t** savedEnvGUIDs)

Saves the environment for the specified user.

parsegraph_removeSavedEnvironments(pool, dbd, int userId, int* numRemoved)

Removes all saved environments for the given user.

parsegraph_saveEnvironment(pool, dbd, int userId, parsegraph_GUID* env, const char* clientSaveState)

Saves the environment for the specified user.
